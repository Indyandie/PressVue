{'/': genSidebarConfig('Pages')}
# Object
function genSidebarConfig (title) {

  // You should put this outisde of module.exports
  return [
    {
      title,
      collapsable: true,
      children: [
        '',
        '/test',
        '/help/vue/',
        '/help/markdown/'
      ]
    }
  ]
}
